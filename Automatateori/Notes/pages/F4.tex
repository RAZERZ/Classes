\section{Övning}
\par\bigskip
\noindent En NFA för språket som beskrivs av $a(aba\cup aa)^*$:\par
\begin{figure}[ht]
    \centering
    \begin{tikzpicture}
      \node[state, initial left, initial text=, ](p0){};
      \node[state, accepting right, right of=p0, xshift=1cm](p1){};
      \path[-stealth] (p0) edge[above] node{$a$} (p1);
      \path[-stealth] (p1) edge[loop above] node{$aba$} (p1);
      \path[-stealth] (p1) edge[loop below] node{$aa$} (p1);
    \end{tikzpicture}
    \caption{}
\end{figure}
\par\bigskip
\noindent Skapa en DFA för samma språk, gör först NFA:n icke-glupsk (och namnge tillstånden):
\begin{figure}[ht]
    \centering
    \begin{tikzpicture}
      \node[state, initial left, initial text=, ](p0){1};
      \node[state, accepting right, right of=p0, xshift=1cm](p1){2};
      \node[state, below of=p1, yshift=-.5cm](p2){$5$};
      \node[state, above of=p1, yshift=.5cm](p3){$3$};
      \node[state, right of=p3, xshift=.5cm](p4){$4$};
      \path[-stealth] (p0) edge[above] node{$a$} (p1);
      \path[-stealth] (p1) edge[right, bend right] node{$a$} (p2);
      \path[-stealth] (p2) edge[right, bend right] node{$a$} (p1);
      \path[-stealth] (p1) edge[left, bend left] node{$a$} (p3);
      \path[-stealth] (p3) edge[below, bend left] node{$b$} (p4);
      \path[-stealth] (p4) edge[below] node{$a$} (p1);
    \end{tikzpicture}
    \caption{}
\end{figure}
